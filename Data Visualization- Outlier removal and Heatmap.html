#Remove outliers from data 
def calculate_outlier(df):
    '''This Function calculates outliers for a given group. One has to inser a dataframe or a list of values to analyse.
    param: Dataframe array 
    return:df_trimmed.to_frame() return Dataframe array without the outliers
    '''
    q1 = df.quantile(.25, interpolation='linear') #calculate quartile1
    q3 = df.quantile(.75, interpolation='linear') #calculate quartile3
    iqr = q3 - q1 #calculate interquartile difference
    lowerbound  = q1 - (1.5 * iqr)  #Lower bound for outliers 
    upperbound  = q3 + (1.5 * iqr)  #upperbound for outliers 
    conditions  = (df >= lowerbound) & (df <= upperbound)
    df_trimmed = df[conditions] #include data without outliers 
    return df_trimmed.to_frame()

def kw_dunn(dataframe,to_compare=compare, alpha=0.05, method='bonf'):
    groups = np.array(([dataframe.loc[type] for type in flyTypes])) 
    """
    Kruskal-Wallis 1-way ANOVA with Dunn's multiple comparison test
    Arguments:
    ---------------
    groups: sequence
        arrays corresponding to k mutually independent samples from
        continuous populations
    to_compare: sequence
        tuples specifying the indices of pairs of groups to compare, e.g.
        [(0, 1), (0, 2)] would compare group 0 with 1 & 2. by default, all
        possible pairwise comparisons between groups are performed.
    alpha: float
        family-wise error rate used for correcting for multiple comparisons
        (see statsmodels.stats.multitest.multipletests for details)
    method: string
        method used to adjust p-values to account for multiple corrections (see
        statsmodels.stats.multitest.multipletests for options)
    Returns:
    ---------------
    H: float
        Kruskal-Wallis H-statistic
    p_omnibus: float
        p-value corresponding to the global null hypothesis that the medians of
        the groups are all equal
    Z_pairs: float array
        Z-scores computed for the absolute difference in mean ranks for each
        pairwise comparison
    p_corrected: float array
        corrected p-values for each pairwise comparison, corresponding to the
        null hypothesis that the pair of groups has equal medians. note that
        these are only meaningful if the global null hypothesis is rejected.
    reject: bool array
        True for pairs where the null hypothesis can be rejected for the given
        alpha
    Reference:
    ---------------
    Gibbons, J. D., & Chakraborti, S. (2011). Nonparametric Statistical
    Inference (5th ed., pp. 353-357). Boca Raton, FL: Chapman & Hall.
    """
    # omnibus test (K-W ANOVA)
    # -------------------------------------------------------------------------


    k = len(groups)

    n = np.array([len(gg) for gg in groups])
    if np.any(n < 5):
        warnings.warn("Sample sizes < 5 are not recommended (K-W test assumes "
                      "a chi square distribution)")
    
    allgroups = np.concatenate(groups)
    N = len(allgroups)
    ranked = stats.rankdata(allgroups)

    # correction factor for ties
    T = stats.tiecorrect(ranked)
    if T == 0:
        raise ValueError('All numbers are identical in kruskal')

    # sum of ranks for each group
    j = np.insert(np.cumsum(n), 0, 0)
    R = np.empty(k, dtype=np.float)
    for ii in range(k):
        R[ii] = ranked[j[ii]:j[ii + 1]].sum()

    # the Kruskal-Wallis H-statistic
    H = (12. / (N * (N + 1.))) * ((R ** 2.) / n).sum() - 3 * (N + 1)

    # apply correction factor for ties
    H /= T

    df_omnibus = k - 1
    p_omnibus = stats.distributions.chi2.sf(H, df_omnibus)

    # multiple comparisons
    # -------------------------------------------------------------------------

    # by default we compare every possible pair of groups
    if to_compare is None:
        to_compare = tuple(combinations(range(k), 2))
        
    ncomp = len(to_compare)
    
    Z_pairs = np.empty(ncomp, dtype=np.float)
    p_uncorrected = np.empty(ncomp, dtype=np.float)
    Rmean = R / n
    
    for pp, (ii, jj) in enumerate(to_compare):
    
    # standardized score
        Zij = (np.abs(Rmean[ii] - Rmean[jj]) / np.sqrt((1. / 12.) * N * (N + 1) * (1. / n[ii] + 1. / n[jj])))
        Z_pairs[pp] = Zij

    # corresponding p-values obtained from upper quantiles of the standard
    # normal distribution
    p_uncorrected = stats.norm.sf(Z_pairs) * 2.

    # correction for multiple comparisons
    reject, p_corrected, alphac_sidak, alphac_bonf = multipletests(p_uncorrected, method=method)
    #return  p_omnibus,p_corrected
    return  p_corrected

def statsflies(dataframe, motor_parameter):
    '''
    This function will perform all statistic analyses including normality and standard deviation tests. 
    If data is normally distributed and homoscedastic (i.e. equal variance for all predictors),
    it will perform a one-way anova; otherwise it will perform a non-parametric test (kruskal-wallis).
    Moreover it will perform multicomparison tests to compare between the data and the control.
    
    :param dataframe: dataframe with data to be analysed organized in NxP N is the fly sample and P the motor parameters 
    :param motor_parameter: str argument that calls the motor variable under consideration
    
    :return pvaluesall: List of pvalues for a specific motor parameter
    
    '''
    pvalues_all = []
    df_selection = dataframe.set_index('Stock')  # we indexed the goup names to the data so we don't lose labels 
    #Select the data that is going to be analysed

    for name_group in flyTypes:
        motor_group.append(df_selection[motor_parameter].loc[name_group])

        groups_out.append(calculate_outlier(motor_group[i]))

    concat_groups = pd.concat(groups_out)
    motor_data=concat_groups[motor_parameter] # resolve some issues regarding labeling of motor parameter

    #create a variable called motor_data_group_label that contains the labels for the groups 
    md_noindex = concat_groups.reset_index()
    motor_data_group_label= md_noindex['Stock'] #Contains only group labels after outlier selection
    ('Data treated: {}'.format(motor_data))  
    
    #Statistical tests 
    # First Test: test for normality - shapiro-wilk test
    shapiro_tests_p = np.array(
        [stats.shapiro(motor_data.loc[type])[1] for type in flyTypes])

    #Second test: test for homoscedasticity - levene test
    levene_test_p = stats.levene(*[motor_data.loc[type]
                                   for type in flyTypes])[1]

    # condition to check that the data is both normal and homoscedastic
    if (len(np.where(shapiro_tests_p > 0.05)[0]) == len(flyTypes)) and (
            levene_test_p > 0.05):
        
        anova_test = stats.mstats.f_oneway(
            *[motor_data.loc[type].values for type in flyTypes])
        anova_test_p = anova_test[1]


    # post-hoc analysis to check which pairs of means are significantly different from each other
        tukey_test = pairwise_tukeyhsd(
            endog=motor_data,  #Tukey test predefined function
            groups=motor_data_group_label,  #fly groups
            alpha=0.05)
        Results = tukey_test.summary()
        tukey_results = pd.DataFrame(
            data=Results.data[1:],

        group1 = tukey_results['group1']         #select first two columns of the table that contain the groups

        control = flyTypes[0]

        mean_diffs = []
        std_pairs = []

                if group == control and group2[i] == x:
                    mean_diffs.append(tukey_test.meandiffs[i])
                    std_pairs.append(tukey_test.std_pairs[i])
                elif group == x and group2[i] == control:
                    mean_diffs.append(tukey_test.meandiffs[i])
                    std_pairs.append(tukey_test.std_pairs[i])

        st_range = np.abs(mean_diffs) / std_pairs
        pvalues = psturng(
            st_range, len(tukey_test.groupsunique[:(len(flyTypes) - 1)]),                   #formula to calculate pvalues
            tukey_test.df_total)


    else:
        kruskal_test = stats.mstats.kruskalwallis(*[motor_data.loc[type].values for type in flyTypes]) #H, pval
        pvalues = kw_dunn(motor_data,
                          to_compare=compare,

def compute_means(dataframe, motor_parameter):
    """
    This function will take all the values of motor_parameter dataframe and compute the means for each group 
    of flies. Important to understand if the parameters are increasing or decreasing 
    
    param dataframe: dataframe with data to be analysed
           motor_parameter: str Name of the motor parameter under consideration
    Return mean_list:  a list of means for each group of flies
    """
    mean_list=[]
    df_selection = dataframe.set_index('Stock')
    #Select the data that is going to be analysed
    motor_group = []
    for name_group in flyTypes:
        motor_group.append(df_selection[motor_parameter].loc[name_group])
    #calculate the outliers for the three groups using outlier function 
    groups_out = []
    for i in range(num_groups):
        groups_out.append(calculate_outlier(motor_group[i]))
    #concatenate groups together in pandas
    concat_groups = pd.concat(groups_out) 
    motor_data=concat_groups[motor_parameter]
    #calculate average for each group
    for type_index in flyTypes:
        mean_group = np.mean(motor_data.loc[type_index].values)
        mean_list.append(mean_group)
    return mean_list

def classify_values(dataframe,motor_parameter):
    """
    classify values function will separate the p_values of each group 4 different bins accoring to the level of significance
    :param dataframe: dataframe with data to be analysed
    :param motor_parameter:str Name of the motor parameter under consideration 
    
    :return pvalues_class: a list of values separating the pvalues in bins accoring to the level of significance
    """

    pvalues_class= []
    pvalues_all_stat = statsflies(dataframe, motor_parameter)
    pvalues_all_array = np.array(pvalues_all_stat)
    pvalues_all=pvalues_all_array[0,:]
    #print(pvalues_all)
    for index in range(len(pvalues_all)):
        if pvalues_all[index] >= 0.05:
            pvalues_class.append(0)
        elif pvalues_all[index] < 0.05 and pvalues_all[index] >= 0.01:
            pvalues_class.append(2.5)
        elif pvalues_all[index]< 0.01 and pvalues_all[index] > 0.001:
             pvalues_class.append(5) 
        elif pvalues_all[index] <= 0.001:
            pvalues_class.append(7.5)
        else:
            pvalues_class.append(0)
    return pvalues_class

def pvalue_mod(dataframe,motor_parameter):
    """
    This function will separate the pvalues into two groups depending if value increases or decreases
    in reation to the control group 
    :param dataframe: dataframe with data to be analysed
    :param motor_parameter:str Name of the motor parameter under consideration 
    
    :return pvalues_mod: list binned pvalues positive or negative depending if groups values increase or decrease in relation
    to the control, respoectively
    """
    pvalue_mod= []
    pvalues_class = classify_values(dataframe,motor_parameter)
    mean_list= compute_means(dataframe, motor_parameter)


    for n in range(len(mean_list)-1):
        if mean_list[0] > mean_list[n+1]: 
            pvalue_mod.append(-1)
        else:
            pvalue_mod.append(1)
        #print (pvalue_mod)
    pvalue_mod = np.array(pvalue_mod)
    pvalues_mod = pvalue_mod * pvalues_class
    return pvalues_mod
